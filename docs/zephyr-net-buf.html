Zephyr net_buf 缓冲区系统

轻量级、引用计数的动态缓冲区管理库

📋 1. 概述

net_buf 是 Zephyr 的核心缓冲区管理库，提供 引用计数、碎片链、内存池分配 等功能。虽然名字带"net"，但它完全不依赖网络协议栈，是通用的 OS 服务组件。

设计目标

零拷贝传递（引用计数）
动态分配和回收
缓冲区碎片链
固定/可变大小数据池
预分配头空间 (reserve)

应用场景

USB 设备栈 (UDC) 数据传输
蓝牙 HCI 数据包
网络协议栈 (TCP/IP)
传感器数据采集
任意生产者-消费者模式
Application
⬇ net_buf_alloc
net_buf_pool (k_lifo free list)
⬇ 分配
net_buf (ref count / frags / user_data)
⬇ data 指针
Data Buffer (uint8_t[])

📦 2. 两种缓冲区模型

net_buf_simple — 轻量栈缓冲
struct net_buf_simple {
    uint8_t  *data;       // 数据起始指针
    uint16_t  len;         // 数据长度
    uint16_t  size;        // 总容量
    uint8_t  *__buf;       // 存储区起始
};

特点：无引用计数、无动态分配、栈上使用。适合一次性、生命周期明确的场景。共 4 个字段，16 字节（32位）或 24 字节（64位）。

net_buf — 完整引用计数缓冲
struct net_buf {
    sys_snode_t  node;       // 链表节点
    struct net_buf  *frags;  // 碎片链表
    uint8_t  ref;            // 引用计数
    uint8_t  flags;
    uint8_t  pool_id;         // 所属池 ID
    uint8_t  user_data_size;
    // ── 与 net_buf_simple ABI 兼容 ──
    uint8_t  *data; uint16_t len;
    uint16_t size; uint8_t *__buf;
    // ── 用户数据 ──
    uint8_t  user_data[];
};

特点：引用计数、碎片链、池管理、用户数据区域。通过 net_buf_pool 动态分配。

🏊 3. 缓冲池 (net_buf_pool)

固定数据大小

NET_BUF_POOL_FIXED_DEFINE(my_pool, 4, 256, 8, NULL);
// 参数: 池名, 缓冲区数量, 数据大小, 用户数据大小, 释放回调

// 展开后分配:
//   - 4 个 net_buf 结构体 (含 8 字节 user_data 每)
//   - 4 x 256 = 1024 字节数据存储区
//   - 1 个 k_lifo 空闲队列

可变数据大小

NET_BUF_POOL_DEFINE(my_pool, 4, 256, 8, NULL);
// 与 FIXED 的区别：数据在堆上动态分配
// 每次 alloc 时 malloc 数据区，free 时释放
// 适合不同大小缓冲区的场景

生命周期

// 1. 定义池（编译期分配内存）
NET_BUF_POOL_FIXED_DEFINE(my_pool, 10, 64, 0, NULL);

// 2. 分配缓冲区
struct net_buf *buf = net_buf_alloc(&my_pool, K_SECONDS(1));
// 如果在中断上下文，只能用 K_NO_WAIT

// 3. 使用缓冲区
net_buf_add_mem(buf, data, len);

// 4. 增加引用计数（共享所有权）
net_buf_ref(buf);

// 5. 释放缓冲区（引用计数归零时回收）
net_buf_unref(buf);

🔧 4. API 速查表

分配与释放

net_buf_alloc(&pool, timeout)从池分配
net_buf_ref(buf)引用 +1
net_buf_unref(buf)引用 -1，归零时回收
net_buf_reset(buf)重置数据指针
net_buf_slist_put(list, buf)放入链表
net_buf_slist_get(list)从链表取出

数据写入

net_buf_add(buf, len)尾部预留 len 字节
net_buf_add_mem(buf, mem, len)尾部复制数据
net_buf_add_u8(buf, val)尾部添加 uint8
net_buf_add_le16(buf, val)尾部添加 LE uint16
net_buf_add_be16(buf, val)尾部添加 BE uint16
net_buf_add_le32(buf, val)尾部添加 LE uint32

数据读取

net_buf_pull(buf, len)从头部取出 len 字节
net_buf_pull_u8(buf)取出 uint8
net_buf_pull_le16(buf)取出 LE uint16
net_buf_remove_mem(buf, len)移除并返回

头部预留

net_buf_headroom(buf)头部剩余空间
net_buf_tailroom(buf)尾部剩余空间
net_buf_push(buf, len)头部预留（反方向）
net_buf_push_u8(buf, data)头部插入 uint8
net_buf_reserve(buf, headroom)初始化时预留头部

碎片链

net_buf_frag_last(buf)获取最后一个碎片
net_buf_frag_add(head, frag)追加碎片到尾部
net_buf_frag_ref(buf)引用整个链
net_buf_frag_unref(buf)释放整个链

查询

net_buf_max_len(buf)缓冲区最大数据长度
net_buf_leftovers(buf)剩余未处理数据
net_buf_simple_is_last(buf)是否为最后一个buffer

⛓️ 5. 碎片链 (Fragment Chain)

碎片链是 net_buf 处理变长数据包的机制。一个 net_buf 的 frags 字段指向链中的下一个，形成单向链表：

  buf1 ───frags──→ buf2 ───frags──→ buf3 ───frags──→ NULL
  ├─ 64 bytes ─┤   ├─ 64 bytes ─┤   ├─ 32 bytes ─┤
  └─── 共 160 字节数据

使用场景：

  • USB 传输：UDC 驱动将大包拆为多个 net_buf 碎片
  • 网络 TCP：一个 socket 数据包可能跨多个缓冲区
  • 蓝牙 ATT：长 MTU 用碎片链承载
// 创建碎片链
struct net_buf *head = net_buf_alloc(&pool, K_NO_WAIT);
net_buf_add_mem(head, "header", 6);

struct net_buf *frag = net_buf_alloc(&pool, K_NO_WAIT);
net_buf_add_mem(frag, "payload data...", 14);

net_buf_frag_add(head, frag);  // 追加到链尾

// 遍历链
struct net_buf *cur = head;
while (cur) {
    process(cur->data, cur->len);
    cur = cur->frags;
}

🔌 6. USB 设备栈中的应用

在 Zephyr USB 设备栈（device_next）中，net_buf 是 UDC 驱动与类驱动之间的数据传输载体：

  USB 中断
    ↓
  udc_stm32.c: 从 USB FIFO 读取数据 → net_buf 填充
    ↓
  usbd_core.c: 事件循环 → usbd_class_request()
    ↓
  类驱动: net_buf->data 读取/写入

UDC 驱动分配 net_buf 的方式：

// usbd_ep_buf_alloc() — 类驱动向 UDC 申请传输缓冲区
struct net_buf *buf = usbd_ep_buf_alloc(c_data, ep, size);
net_buf_add_mem(buf, data, len);
usbd_ep_enqueue(c_data, buf);   // 提交传输

// 传输完成后收到回调
static int my_request_handler(struct usbd_class_data *c_data,
                             struct net_buf *buf, const int err) {
    // buf->data, buf->len 包含 USB 接收的数据
    process_data(buf->data, buf->len);
    net_buf_unref(buf);  // 必须释放！
}

UDC 驱动使用 net_buf_pool 管理缓冲区，通过 udc_buf_pool （定义在 drivers/usb/common/buf/）提供统一的池接口。

📝 7. 完整示例程序

生产者-消费者模式

/* prj.conf */
CONFIG_NET_BUF=y
CONFIG_NET_BUF_POOL_USAGE=y
/* main.c — net_buf 生产者-消费者示例 */
#include <zephyr/kernel.h>
#include <zephyr/net_buf.h>
#include <zephyr/sys/printk.h>

/* 定义缓冲区池：10 个缓冲区，每个 128 字节，4 字节用户数据 */
NET_BUF_POOL_FIXED_DEFINE(my_pool, 10, 128, 4, NULL);

/* 数据链表 — 用于线程间传递 */
static struct k_slist data_list;

/* 生产者线程 */
void producer_thread(void *arg1, void *arg2, void *arg3) {
    uint32_t counter = 0;

    while (1) {
        /* 从池中分配缓冲区 */
        struct net_buf *buf = net_buf_alloc(&my_pool, K_SECONDS(1));
        if (!buf) { printk("alloc timeout\n"); continue; }

        /* 构造数据包 */
        net_buf_add_mem(buf, "Packet #", 8);
        net_buf_add_le32(buf, counter++);
        net_buf_add(buf, 1);  // 终止符预留

        /* 写入用户数据（元信息） */
        *(uint32_t *)net_buf_user_data(buf) = counter;

        /* 放入链表传递 */
        net_buf_slist_put(&data_list, buf);
        printk("Produced: len=%u, id=%u\n",
               buf->len, counter);

        k_sleep(K_SECONDS(2));
    }
}

/* 消费者线程 */
void consumer_thread(void *arg1, void *arg2, void *arg3) {
    while (1) {
        /* 从链表获取缓冲区（阻塞） */
        struct net_buf *buf = net_buf_slist_get(&data_list);
        if (!buf) { k_sleep(K_MSEC(100)); continue; }

        /* 读取用户数据 */
        uint32_t id = *(uint32_t *)net_buf_user_data(buf);

        /* 处理数据 */
        printk("Consumed: len=%u, id=%u, data=",
               buf->len, id);
        for (int i = 0; i < buf->len; i++) {
            printk("%02x ", buf->data[i]);
        }
        printk("\n");

        /* 释放缓冲区（引用计数归零时自动回池） */
        net_buf_unref(buf);
    }
}

K_THREAD_DEFINE(producer_id, 1024,
                producer_thread, NULL, NULL, NULL,
                5, 0, 0);
K_THREAD_DEFINE(consumer_id, 1024,
                consumer_thread, NULL, NULL, NULL,
                5, 0, 0);

int main(void) {
    k_slist_init(&data_list);
    printk("net_buf producer/consumer example\n");
    return 0;
}

⚡ 8. 高级用法

引用计数共享

// 同一块数据被多个消费者共享
struct net_buf *buf = net_buf_alloc(&pool, K_NO_WAIT);
net_buf_add_mem(buf, data, len);

struct net_buf *clone = buf;
net_buf_ref(clone);  // ref=2，底层数据不复制

// 两个指针独立操作 data 偏移，但共享同一存储
// 各自调用 net_buf_unref()，最后一个释放时回收

头部预留 (Reserve/ Push)

// 协议栈分层处理时预留协议头空间
struct net_buf *buf = net_buf_alloc(&pool, K_NO_WAIT);
net_buf_reserve(buf, 4);  // 预留 4 字节头部

// 下层协议填充 payload
net_buf_add_mem(buf, payload, len);

// 上层协议插入头部
net_buf_push_u8(buf, flags);
net_buf_push_le16(buf, type);
net_buf_push_u8(buf, version);
